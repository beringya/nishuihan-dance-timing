```python
def extract_gold_components(frame):
    hsv = cvtColor(frame, BGR2HSV)
    mask = inRange(hsv, lower_gold, upper_gold)
    mask = morphology_open(mask, kernel=3)
    mask = morphology_close(mask, kernel=5)

    contours = findContours(mask)
    components = []
    for c in contours:
        x, y, w, h = boundingRect(c)
        area = contourArea(c)
        if area < circle_min_area:

        components.append(Component(x, y, w, h, area))
```

## 3. 短按时机分数

```python
def compute_tap_alignment_score(button, ring):
    center_delta = distance(button.center, ring.center)
    radius_delta = abs(button.radius - ring.radius)

    center_score = clamp01(1 - center_delta / center_tolerance)
    radius_score = clamp01(1 - radius_delta / radius_tolerance)

    return 0.55 * center_score + 0.45 * radius_score
```

如果暂时无法稳定提取外圈，可以采用保底策略：

## 5. 轨迹预测

用于提前触发，抵消系统延迟。

```python
class MotionPredictor:
    def update(self, position, time_ms):
        self.history.append((position, time_ms))
        self.history = self.history[-5:]

    def predict_ms_to_target(self, target_position):
        if len(self.history) < 2:
            return None
        p0, t0 = self.history[-2]
        p1, t1 = self.history[-1]
        v = (p1 - p0) / max(t1 - t0, 1e-3)
        if abs(v) < 1e-3:
            return None
        return (target_position - p1) / v
```

```python

    if predicted_ms_to_hit <= input_advance_ms:
        trigger_now()
```

### 6.1 短按

```python
if state == "Idle":
    if tap_score >= tap_threshold for N consecutive frames:
        Tap(key, down_ms=35)
        state = "Cooldown"

elif state == "Cooldown":
```

### 6.2 长按

```python
if state == "Idle":
    if hold_start_score >= down_threshold for N consecutive frames:
        KeyDown(key)
        state = "Holding"

elif state == "Holding":
    if hold_tail_score >= up_threshold:
        KeyUp(key)
        state = "Cooldown"
    elif prompt_missing_frames > max_missing_frames:
        KeyUp(key)
        state = "Cooldown"

elif state == "Cooldown":
    if start_score <= reset_threshold and tail_score <= reset_threshold:
        state = "Idle"
```
